1//! Kumiho Reference (Kref) — a URI uniquely identifying any Kumiho object.
2//!
3//! Format: `kref://project/space/item.kind?r=REVISION&a=ARTIFACT`
4//!
5//! The validation rules here mirror the Python gold standard exactly so that
6//! all SDKs accept and reject the same set of URIs.
7
8use crate::pb;
9use once_cell::sync::Lazy;
10use regex::Regex;
11use std::fmt;
12
13/// Raised when a kref URI is malformed or contains a malicious pattern.
14#[derive(Debug, Clone, thiserror::Error)]
15#[error("{0}")]
16pub struct KrefValidationError(pub String);
17
18// Path segments may contain Unicode letters/numbers plus underscore, dots and
19// hyphens. Python's `re` `\w` matches exactly `[\p{L}\p{N}_]`; the regex crate's
20// `\w` differs (it also matches combining marks `\p{M}` and connector punctuation
21// `\p{Pc}`, and uses `\p{Nd}` so it excludes other-numbers `\p{No}`), so we spell
22// the class out to match the Python gold standard. Artifact ids stay ASCII because
23// they are server-generated opaque identifiers. Path traversal (`..`) and control
24// characters are rejected by explicit checks, not by this pattern.
25static KREF_PATTERN: Lazy<Regex> = Lazy::new(|| {
26    Regex::new(
27        r"^kref://(/[\p{L}\p{N}_][\p{L}\p{N}_.\-]*(/[\p{L}\p{N}_][\p{L}\p{N}_.\-]*)*|[\p{L}\p{N}_][\p{L}\p{N}_.\-]*(/[\p{L}\p{N}_][\p{L}\p{N}_.\-]*)*)(\?r=\d+(&a=[a-zA-Z0-9._\-]+)?)?$",
28    )
29    .expect("kref regex is valid")
30});
31
32/// Validate a kref URI for security and correctness.
33///
34/// Checks the `kref://` scheme, rejects path traversal (`..`) and control
35/// characters, and enforces the path/query grammar.
36pub fn validate_kref(uri: &str) -> Result<(), KrefValidationError> {
37    if uri.contains("..") {
38        return Err(KrefValidationError(format!(
39            "Invalid kref URI '{uri}': path traversal (..) not allowed"
40        )));
41    }
42    if uri.chars().any(|c| (c as u32) < 32 || c == '\u{7f}') {
43        return Err(KrefValidationError(format!(
44            "Invalid kref URI '{uri}': control characters not allowed"
45        )));
46    }
47    if !KREF_PATTERN.is_match(uri) {
48        return Err(KrefValidationError(format!(
49            "Invalid kref URI '{uri}': must be format kref://project/space/item.kind"
50        )));
51    }
52    Ok(())
53}
54
55/// Returns `true` if `uri` is a valid kref, without allocating an error.
56pub fn is_valid_kref(uri: &str) -> bool {
57    validate_kref(uri).is_ok()
58}

60/// A validated Kumiho reference.
61///
62/// `Kref` is a thin wrapper around a `String`; it derefs to `str` and displays
63/// as the underlying URI, so it can be used anywhere a string is expected.
64#[derive(Debug, Clone, PartialEq, Eq, Hash, PartialOrd, Ord)]
65pub struct Kref(String);
66
67impl Kref {
68    /// Parse and validate a kref URI.
69    pub fn new(uri: impl Into<String>) -> Result<Self, KrefValidationError> {
70        let uri = uri.into();
71        validate_kref(&uri)?;
72        Ok(Kref(uri))
73    }
74
75    /// Wrap a string as a kref **without** validation.
76    ///
77    /// Use only for trusted, server-returned values (the server may emit krefs
78    /// whose shape predates current client validation, e.g. Unicode ids).
79    pub fn unchecked(uri: impl Into<String>) -> Self {
80        Kref(uri.into())
81    }
82
83    /// Build a [`Kref`] from a protobuf `Kref` message (trusted, unvalidated).
84    pub fn from_pb(pb: &pb::Kref) -> Self {
85        Kref(pb.uri.clone())
86    }
87
88    /// Convert into a protobuf `Kref` message for gRPC requests.
89    pub fn to_pb(&self) -> pb::Kref {
90        pb::Kref {
91            uri: self.0.clone(),
92        }
93    }
94
95    /// The full URI string.
96    pub fn uri(&self) -> &str {
97        &self.0
98    }
99
100    /// The path component (after `kref://`, before any `?` query).
101    pub fn path(&self) -> &str {
102        let after = self.0.split_once("://").map(|x| x.1).unwrap_or(&self.0);
103        after.split('?').next().unwrap_or(after)
104    }
105
106    /// The project name (first path segment).
107    pub fn project(&self) -> &str {
108        let p = self.path();
109        p.split_once('/').map(|x| x.0).unwrap_or(p)
110    }
111
112    /// The space path (segments between project and item), or `""` if none.
113    pub fn space(&self) -> String {
114        let p = self.path();
115        let parts: Vec<&str> = p.split('/').collect();
116        if parts.len() <= 2 {
117            return String::new();
118        }
119        parts[1..parts.len() - 1].join("/")
120    }
121
122    /// The item name including kind (e.g. `hero.model`), or `""` if none.
123    pub fn item_name(&self) -> &str {
124        let p = self.path();
125        match p.rsplit_once('/') {
126            Some((_, last)) => last,
127            None => "",
128        }
129    }
130
131    /// The item kind (after the first `.` in the item name), or `""`.
132    pub fn kind(&self) -> &str {
133        let name = self.item_name();
134        name.split_once('.').map(|x| x.1).unwrap_or("")
135    }
136
137    /// The revision number from `?r=`, defaulting to `1`.
138    pub fn revision(&self) -> i32 {
139        self.0
140            .split_once("?r=")
141            .and_then(|(_, rest)| {
142                let digits: String = rest.chars().take_while(|c| c.is_ascii_digit()).collect();
143                digits.parse().ok()
144            })
145            .unwrap_or(1)
146    }
147
148    /// The artifact name from `&a=`, if present.
149    pub fn artifact_name(&self) -> Option<String> {
150        self.0
151            .split_once("&a=")
152            .map(|(_, rest)| rest.chars().take_while(|&c| c != '&').collect())
153    }
154}
155
156impl fmt::Display for Kref {
157    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
158        f.write_str(&self.0)
159    }
160}
161
162impl std::ops::Deref for Kref {
163    type Target = str;
164    fn deref(&self) -> &str {
165        &self.0
166    }
167}
168
169impl std::str::FromStr for Kref {
170    type Err = KrefValidationError;
171    fn from_str(s: &str) -> Result<Self, Self::Err> {
172        Kref::new(s)
173    }
174}
175
176impl From<Kref> for String {
177    fn from(k: Kref) -> String {
178        k.0
179    }
180}
181
182impl PartialEq<str> for Kref {
183    fn eq(&self, other: &str) -> bool {
184        self.0 == other
185    }
