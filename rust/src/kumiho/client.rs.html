31const DEFAULT_RPC_TIMEOUT_SECS: f64 = 30.0;
32const RETRY_MAX_ATTEMPTS: u32 = 3;
33const RETRY_BASE_DELAY_MS: u64 = 500;
34const RETRY_MAX_DELAY_MS: u64 = 5000;
35
36fn retry_max_attempts() -> u32 {
37    std::env::var("KUMIHO_GRPC_RETRY_MAX_ATTEMPTS")
38        .ok()
39        .and_then(|v| v.parse().ok())
40        .unwrap_or(RETRY_MAX_ATTEMPTS)
41}
42
43fn rpc_timeout() -> Option<Duration> {
44    let secs = std::env::var("KUMIHO_RPC_TIMEOUT_SECS")
45        .ok()
46        .and_then(|v| v.parse::<f64>().ok())
47        .unwrap_or(DEFAULT_RPC_TIMEOUT_SECS);
48    if secs <= 0.0 {
49        None
50    } else {
51        Some(Duration::from_secs_f64(secs))
52    }
53}
54
55fn backoff(attempt: u32) -> Duration {
56    let base = RETRY_BASE_DELAY_MS * (1u64 << (attempt - 1).min(16));
57    let capped = base.min(RETRY_MAX_DELAY_MS);
58    let jitter = (rand::random::<f64>() * (capped as f64) * 0.25) as u64;
59    Duration::from_millis(capped + jitter)
60}

62fn is_transient(code: tonic::Code) -> bool {
63    matches!(
64        code,
65        tonic::Code::Unavailable
66            | tonic::Code::DeadlineExceeded
67            | tonic::Code::Internal
68            | tonic::Code::ResourceExhausted
69    )
70}

72/// Runs a unary RPC with transient-failure retry + per-call deadline.
73macro_rules! unary {
74    ($self:ident, $method:ident, $msg:expr) => {{
75        let msg = $msg;
76        let max = $self.max_attempts;
77        let mut attempt: u32 = 0;
78        loop {
79            let mut grpc = $self.grpc.clone();
80            let mut request = tonic::Request::new(msg.clone());
81            if let Some(d) = $self.rpc_timeout {
82                request.set_timeout(d);
83            }
84            match grpc.$method(request).await {
85                Ok(r) => break Ok::<_, crate::Error>(r.into_inner()),
86                Err(status) => {
87                    attempt += 1;
88                    if is_transient(status.code()) && attempt < max {
89                        tokio::time::sleep(backoff(attempt)).await;
90                        continue;
91                    }
92                    break Err(crate::Error::Rpc(status));
93                }
94            }
95        }
96    }};
97}

101#[derive(Clone)]
102pub(crate) struct KumihoInterceptor {
103    metadata: Vec<(String, String)>,
104}
105
106impl tonic::service::Interceptor for KumihoInterceptor {
107    fn call(
108        &mut self,
109        mut req: tonic::Request<()>,
110    ) -> std::result::Result<tonic::Request<()>, tonic::Status> {
111        let md = req.metadata_mut();
112        for (k, v) in &self.metadata {
113            if let (Ok(key), Ok(val)) = (
114                AsciiMetadataKey::from_bytes(k.as_bytes()),
115                AsciiMetadataValue::try_from(v.as_str()),
116            ) {
117                md.insert(key, val);
118            }
119        }
120        let cid = format!("kumiho-{}", uuid::Uuid::new_v4().simple());
121        if let Ok(val) = AsciiMetadataValue::try_from(cid.as_str()) {
122            md.insert("x-correlation-id", val);
123        }
124        Ok(req)
125    }
126}

128/// A page of list results carrying an optional pagination cursor.
129#[derive(Debug, Clone)]
130pub struct Page<T> {
131    /// The items on this page.
132    pub items: Vec<T>,
133    /// Cursor for the next page, if more results exist.
134    pub next_cursor: Option<String>,
135    /// Total count across all pages, if the server reported it.
136    pub total_count: Option<i32>,
137}

139impl<T> std::ops::Deref for Page<T> {
140    type Target = [T];
141    fn deref(&self) -> &[T] {
142        &self.items
143    }
144}

146impl<T> IntoIterator for Page<T> {
147    type Item = T;
148    type IntoIter = std::vec::IntoIter<T>;
149    fn into_iter(self) -> Self::IntoIter {
150        self.items.into_iter()
151    }
152}

154fn page_from<T>(items: Vec<T>, pagination: Option<pb::PaginationResponse>) -> Page<T> {
155    match pagination {
156        Some(p) => Page {
157            items,
158            next_cursor: (!p.next_cursor.is_empty()).then_some(p.next_cursor),
159            total_count: Some(p.total_count),
160        },
161        None => Page {
162            items,
163            next_cursor: None,
164            total_count: None,
165        },
166    }
167}

257    /// Resolve routing/auth and build the [`Client`].
258    pub async fn build(mut self) -> Result<Client> {
259        // 1. Resolve token.
260        let mut token = match self.token.take() {
261            Some(t) => Some(t),
262            None => crate::token_loader::load_bearer_token().map_err(Error::InvalidArgument)?,
263        };
264
265        // 2. No endpoint + no token -> try a local self-hosted CE server.
266        if self.endpoint.is_none() && token.is_none() {
267            if let Some(local) = crate::discovery::resolve_local_ce_endpoint().await? {
268                self.endpoint = Some(local);
269                self.use_discovery = Some(false);
270            }
271        }
272
273        let mut metadata = self.metadata.clone();
274
275        // 3. Discovery (when no explicit endpoint, enabled, token present).
276        let discovery_enabled = self
277            .use_discovery
278            .unwrap_or_else(|| !env_flag("KUMIHO_DISABLE_AUTO_DISCOVERY"));
279        if self.endpoint.is_none() && discovery_enabled {
280            if let Some(tok) = token.clone() {
281                match crate::discovery::resolve(
282                    &tok,
283                    self.tenant_hint.as_deref(),
284                    self.control_plane_url.as_deref(),
285                    self.cache_path.as_deref(),
286                    self.force_discovery_refresh,
287                )
288                .await
289                {
290                    Ok(record) => {
291                        self.endpoint = Some(record.target());
292                        metadata.push(("x-tenant-id".into(), record.tenant_id.clone()));
293                    }
294                    Err(_) => {
295                        if let Some(hint) = &self.tenant_hint {
296                            metadata.push(("x-tenant-id".into(), hint.clone()));
297                        }
298                    }
299                }
300            } else if let Some(hint) = &self.tenant_hint {
301                metadata.push(("x-tenant-id".into(), hint.clone()));
302            }
303        }
304
305        // 4. Fall back to env / localhost.
306        let target = self.endpoint.clone().unwrap_or_else(|| {
307            std::env::var("KUMIHO_SERVER_ENDPOINT")
308                .or_else(|_| std::env::var("KUMIHO_SERVER_ADDRESS"))
309                .unwrap_or_else(|_| "localhost:8080".to_string())
310        });
311
312        // 5. Normalize + build channel.
313        let (host, port, mut use_tls) = normalize_target(&target)?;
314        if let Ok(v) = std::env::var("KUMIHO_SERVER_USE_TLS") {
315            use_tls = matches!(v.to_ascii_lowercase().as_str(), "1" | "true" | "yes");
316        }
317        let channel = build_channel(&host, port, use_tls).await?;
318
319        // 6. Auth header.
320        if let Some(tok) = token.take() {
321            metadata.push(("authorization".into(), format!("Bearer {tok}")));
322        }
323
324        let interceptor = KumihoInterceptor { metadata };
325        let grpc =
326            pb::kumiho_service_client::KumihoServiceClient::with_interceptor(channel, interceptor)
327                .max_decoding_message_size(64 * 1024 * 1024);
328
329        Ok(Client {
330            grpc,
331            max_attempts: retry_max_attempts(),
332            rpc_timeout: rpc_timeout(),
333        })
334    }
335}

337// Mirrors the Python _Client._env_flag used for KUMIHO_DISABLE_AUTO_DISCOVERY:
338// an unset variable is false, but any set value other than 0/false/no (including
339// an empty string) is true.
340fn env_flag(name: &str) -> bool {
341    std::env::var(name)
342        .map(|v| !matches!(v.trim().to_ascii_lowercase().as_str(), "0" | "false" | "no"))
343        .unwrap_or(false)
344}
345
346fn normalize_target(raw: &str) -> Result<(String, u16, bool)> {
347    let target = raw.trim();
348    if target.is_empty() {
349        return Err(Error::InvalidArgument("endpoint cannot be empty".into()));
350    }
351    let (scheme, rest) = match target.split_once("://") {
352        Some((s, r)) => (s.to_ascii_lowercase(), r),
353        None => (String::new(), target),
354    };
355    let hostport = rest.split('/').next().unwrap_or(rest);
356    let (host, port_opt) = match hostport.rsplit_once(':') {
357        Some((h, p)) => (h.to_string(), p.parse::<u16>().ok()),
358        None => (hostport.to_string(), None),
359    };
360    if host.is_empty() {
361        return Err(Error::InvalidArgument(format!("invalid endpoint: {raw}")));
362    }
363    let tls_scheme = matches!(scheme.as_str(), "https" | "grpcs");
364    let port = port_opt.unwrap_or_else(|| match scheme.as_str() {
365        "https" | "grpcs" => 443,
366        "http" | "grpc" => 80,
367        _ => 8080,
368    });
369    let use_tls = tls_scheme || port == 443;
370    Ok((host, port, use_tls))
371}
372
373async fn build_channel(host: &str, port: u16, use_tls: bool) -> Result<Channel> {
374    let scheme = if use_tls { "https" } else { "http" };
375    let url = format!("{scheme}://{host}:{port}");
376    let mut endpoint = Endpoint::from_shared(url)
377        .map_err(|e| Error::InvalidArgument(format!("invalid endpoint: {e}")))?
378        .http2_keep_alive_interval(Duration::from_secs(30))
379        .keep_alive_timeout(Duration::from_secs(10))
380        .keep_alive_while_idle(true);
381
382    if use_tls {
383        let authority =
384            std::env::var("KUMIHO_SERVER_AUTHORITY").unwrap_or_else(|_| host.to_string());
385        let mut tls = ClientTlsConfig::new()
386            .with_native_roots()
387            .domain_name(authority);
388        if let Ok(ca_path) = std::env::var("KUMIHO_SERVER_CA_FILE") {
389            if !ca_path.is_empty() {
390                let pem = std::fs::read(&ca_path)?;
391                tls = tls.ca_certificate(tonic::transport::Certificate::from_pem(pem));
392            }
393        }
394        endpoint = endpoint.tls_config(tls)?;
395    }
396    Ok(endpoint.connect_lazy())
397}

419    }

421    /// Auto-configure following the standard Kumiho bootstrap chain:
422    ///
423    /// 1. Load a bearer token (`KUMIHO_AUTH_TOKEN`, else `~/.kumiho/kumiho_authentication.json`).
424    /// 2. **Token present** → control-plane discovery resolves the tenant's
425    ///    regional cloud kumiho-server (errors propagate).
426    /// 3. **No token** → probe the loopback self-hosted CE server and use it.
427    /// 4. Neither available → returns an error.
428    ///
429    /// For explicit endpoints or a localhost dev fallback, use
430    /// [`Client::connect`] or [`Client::builder`].
431    pub async fn auto() -> Result<Client> {
432        Self::auto_with_tenant(None).await
433    }
434
435    /// Like [`Client::auto`], but pins discovery to a specific tenant slug/id.
436    pub async fn auto_with_tenant(tenant_hint: Option<&str>) -> Result<Client> {
437        let token = crate::token_loader::load_bearer_token().map_err(Error::InvalidArgument)?;
438        if let Some(tok) = token {
439            // Token present -> control-plane discovery -> tenant's cloud server.
440            let record = crate::discovery::resolve(&tok, tenant_hint, None, None, false).await?;
441            return ClientBuilder::default()
442                .endpoint(record.target())
443                .token(tok)
444                .use_discovery(false)
445                .metadata("x-tenant-id", record.tenant_id)
446                .build()
447                .await;
448        }
449        // No token -> fall back to a local self-hosted CE server if present.
450        if let Some(local) = crate::discovery::resolve_local_ce_endpoint().await? {
451            return ClientBuilder::default()
452                .endpoint(local)
453                .use_discovery(false)
454                .build()
455                .await;
456        }
457        Err(Error::Discovery(
458            "no credentials found: set KUMIHO_AUTH_TOKEN or run `kumiho-cli login`; \
459             no local self-hosted CE server detected on loopback"
460                .into(),
461        ))
462    }

610    /// Create an item. Rejects the reserved `bundle` kind (use [`Client::create_bundle`]).
611    pub async fn create_item(
612        &self,
613        parent_path: &str,
614        item_name: &str,
615        kind: &str,
616        metadata: Option<HashMap<String, String>>,
617    ) -> Result<Item> {
618        if RESERVED_KINDS.contains(&kind.to_ascii_lowercase().as_str()) {
619            return Err(Error::ReservedKind(kind.to_string()));
620        }
621        let req = pb::CreateItemRequest {
622            parent_path: parent_path.to_string(),
623            item_name: item_name.to_string(),
624            kind: kind.to_string(),
625            exists_error: false,
626        };
627        let resp = unary!(self, create_item, req)?;
628        let item = Item::from_pb(resp, self.clone());
629        if let Some(md) = metadata {
630            if !md.is_empty() {
631                return self.update_item_metadata(&item.kref, md).await;
632            }
633        }
634        Ok(item)
635    }

825    pub async fn get_revision(&self, kref_uri: &str) -> Result<Revision> {
826        let (base, tag, time) = parse_tag_time(kref_uri)?;
827        if tag.is_some() || time.is_some() {
828            let req = pb::ResolveKrefRequest {
829                kref: base,
830                tag,
831                time,

834            return Ok(Revision::from_pb(resp, self.clone()));
835        }
836        let req = pb::KrefRequest {
837            kref: Some(pb::Kref {
838                uri: kref_uri.to_string(),
839            }),
840        };
841        let resp = unary!(self, get_revision, req)?;
842        Ok(Revision::from_pb(resp, self.clone()))
843    }

1059    /// Get an artifact by kref URI; falls back to the revision's default artifact.
1060    pub async fn get_artifact_by_kref(&self, kref_uri: &str) -> Result<Artifact> {
1061        let kref = Kref::new(kref_uri).map_err(|e| Error::KrefValidation(e.0))?;
1062        if let Some(name) = kref.artifact_name() {
1063            let revision_uri = kref_uri.split("&a=").next().unwrap_or(kref_uri);
1064            let revision_kref = Kref::new(revision_uri).map_err(|e| Error::KrefValidation(e.0))?;
1065            return self.get_artifact(&revision_kref, &name).await;
1066        }
1067        let revision = self.get_revision(kref_uri).await?;
1068        match &revision.default_artifact {
1069            Some(default_name) => self.get_artifact(&revision.kref, default_name).await,
1070            None => Err(Error::InvalidArgument(format!(
1071                "artifact kref '{kref_uri}' missing &a= and no default_artifact set"
1072            ))),
1073        }
1074    }

1260        edge_type_filter: Option<Vec<String>>,
1261        max_depth: i32,
1262        limit: i32,
1263        include_path: bool,
1264    ) -> Result<TraversalResult> {
1265        let req = pb::TraverseEdgesRequest {
1266            origin_kref: Some(origin_kref.to_pb()),
1267            direction: direction.as_pb(),
1268            edge_type_filter: edge_type_filter.unwrap_or_default(),
1269            max_depth,
1270            limit,
1271            include_path,
1272        };
1273        let resp = unary!(self, traverse_edges, req)?;
1274        let revision_krefs = resp
1275            .revision_krefs
1276            .into_iter()
1277            .map(|k| Kref::unchecked(k.uri))
1278            .collect();
1279        let paths = resp.paths.into_iter().map(map_path).collect();
1280        let edges = resp
1281            .edges
1282            .into_iter()
1283            .map(|e| Edge::from_pb(e, self.clone()))
1284            .collect();
1285        Ok(TraversalResult::new(
1286            revision_krefs,
1287            paths,
1288            edges,
1289            resp.total_count,
1290            resp.truncated,
1291            self.clone(),
1292        ))
1293    }

1544/// Split an item kref URI into `(parent_path_with_leading_slash, name, kind)`.
1545fn split_item_kref(kref_uri: &str) -> Result<(String, String, String)> {
1546    let kref = Kref::new(kref_uri).map_err(|e| Error::KrefValidation(e.0))?;
1547    let path = kref.path();
1548    let (space_path, item_name_kind) = path
1549        .split_once('/')
1550        .ok_or_else(|| Error::InvalidArgument(format!("invalid item kref: {kref_uri}")))?;
1551    let (name, kind) = item_name_kind.split_once('.').ok_or_else(|| {
1552        Error::InvalidArgument(format!("invalid item name.kind: {item_name_kind}"))
1553    })?;
1554    Ok((format!("/{space_path}"), name.to_string(), kind.to_string()))
1555}

1557/// Parse `?t=`/`?tag=`/`?time=` query params from a kref; returns `(base, tag, time)`.
1558fn parse_tag_time(kref_uri: &str) -> Result<(String, Option<String>, Option<String>)> {
1559    let Some((base, params)) = kref_uri.split_once('?') else {
1560        return Ok((kref_uri.to_string(), None, None));
1561    };
1562    let mut tag = None;
1563    let mut time = None;
1564    for param in params.split('&') {
1565        if let Some(v) = param
1566            .strip_prefix("t=")
1567            .or_else(|| param.strip_prefix("tag="))
1568        {
1569            tag = Some(v.to_string());
1570        } else if let Some(v) = param.strip_prefix("time=") {
1571            if v.len() != 12 || !v.chars().all(|c| c.is_ascii_digit()) {
1572                return Err(Error::InvalidArgument(
1573                    "time must be in YYYYMMDDHHMM format".into(),
1574                ));
1575            }
1576            time = Some(v.to_string());
1577        }
1578    }
1579    Ok((base.to_string(), tag, time))
