1//! Locating bearer tokens for gRPC auth (mirrors Python `_token_loader.py`).
2//!
3//! Resolution order:
4//! 1. `KUMIHO_AUTH_TOKEN` env var.
5//! 2. `~/.kumiho/kumiho_authentication.json` — preferring the firebase
6//!    `id_token` unless `KUMIHO_USE_CONTROL_PLANE_TOKEN` is set.
7
8use std::path::PathBuf;
9
10const TOKEN_ENV: &str = "KUMIHO_AUTH_TOKEN";
11const FIREBASE_TOKEN_ENV: &str = "KUMIHO_FIREBASE_ID_TOKEN";
12const USE_CP_TOKEN_ENV: &str = "KUMIHO_USE_CONTROL_PLANE_TOKEN";
13const CREDENTIALS_FILENAME: &str = "kumiho_authentication.json";
14
15fn normalize(value: Option<String>) -> Option<String> {
16    value
17        .map(|v| v.trim().to_string())
18        .filter(|v| !v.is_empty())
19}
20
21fn env_flag(name: &str) -> bool {
22    std::env::var(name)
23        .map(|v| matches!(v.trim().to_ascii_lowercase().as_str(), "1" | "true" | "yes"))
24        .unwrap_or(false)
25}

27/// The Kumiho config directory (`$KUMIHO_CONFIG_DIR` or `~/.kumiho`).
28pub fn config_dir() -> PathBuf {
29    if let Ok(dir) = std::env::var("KUMIHO_CONFIG_DIR") {
30        if !dir.is_empty() {
31            return PathBuf::from(shellexpand_tilde(&dir));
32        }
33    }
34    let home = dirs::home_dir().unwrap_or_else(|| PathBuf::from("."));
35    home.join(".kumiho")
36}
37
38fn shellexpand_tilde(p: &str) -> String {
39    if let Some(rest) = p.strip_prefix("~") {
40        if let Some(home) = dirs::home_dir() {
41            return format!("{}{}", home.display(), rest);
42        }
43    }
44    p.to_string()
45}
46
47fn credentials_path() -> PathBuf {
48    config_dir().join(CREDENTIALS_FILENAME)
49}
50
51fn read_credentials() -> Option<serde_json::Value> {
52    let path = credentials_path();
53    let text = std::fs::read_to_string(path).ok()?;
54    serde_json::from_str(&text).ok()
55}
56
57fn credentials_tokens() -> (Option<String>, Option<String>) {
58    match read_credentials() {
59        Some(data) => {
60            let cp = normalize(
61                data.get("control_plane_token")
62                    .and_then(|v| v.as_str())
63                    .map(String::from),
64            );
65            let id = normalize(
66                data.get("id_token")
67                    .and_then(|v| v.as_str())
68                    .map(String::from),
69            );
70            (cp, id)
71        }
72        None => (None, None),
73    }
74}

76/// Validate a token has JWT shape (`header.payload.signature`).
77pub fn validate_token_format(
78    token: Option<String>,
79    source: &str,
80) -> Result<Option<String>, String> {
81    let token = match normalize(token) {
82        Some(t) => t,
83        None => return Ok(None),
84    };
85    let parts: Vec<&str> = token.split('.').collect();
86    if parts.len() != 3 {
87        return Err(format!(
88            "Invalid {source} format: expected JWT with 3 parts, got {}. Run `kumiho-cli login`.",
89            parts.len()
90        ));
91    }
92    if parts.iter().any(|p| p.is_empty()) {
93        return Err(format!(
94            "Invalid {source} format: a JWT part is empty (token corrupted/incomplete)."
95        ));
96    }
97    Ok(Some(token))
98}

100/// The preferred bearer token for gRPC calls, if any can be found.
101pub fn load_bearer_token() -> Result<Option<String>, String> {
102    if let Some(env_token) = normalize(std::env::var(TOKEN_ENV).ok()) {
103        return validate_token_format(Some(env_token), "KUMIHO_AUTH_TOKEN");
104    }
105    let prefer_cp = env_flag(USE_CP_TOKEN_ENV);
106    let (cp, firebase) = credentials_tokens();
107    if prefer_cp {
108        if let Some(cp) = cp.clone() {
109            return validate_token_format(Some(cp), "control_plane_token");
110        }
111    }
112    if let Some(fb) = firebase {
113        return validate_token_format(Some(fb), "id_token");
114    }
115    if let Some(cp) = cp {
116        return validate_token_format(Some(cp), "control_plane_token");
117    }
118    Ok(None)
119}

121/// A Firebase ID token for control-plane interactions, if available.
122pub fn load_firebase_token() -> Option<String> {
123    if let Some(env_token) = normalize(std::env::var(FIREBASE_TOKEN_ENV).ok()) {
124        return Some(env_token);
125    }
126    credentials_tokens().1
