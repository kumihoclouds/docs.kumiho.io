11/// Raised when an edge type is malformed.
12#[derive(Debug, Clone, thiserror::Error)]
13#[error("{0}")]
14pub struct EdgeTypeValidationError(pub String);
15
16// Must match the Rust server's validation: uppercase start, 1-50 chars.
17static EDGE_TYPE_PATTERN: Lazy<Regex> =
18    Lazy::new(|| Regex::new(r"^[A-Z][A-Z0-9_]{0,49}$").expect("edge type regex is valid"));
19
20/// Validate an edge type (uppercase, `[A-Z0-9_]`, 1-50 chars).
21///
22/// Returns the named [`EdgeTypeValidationError`] (mirroring `validate_kref` and
23/// the Python/Go SDKs); it converts to [`crate::Error`] via `?`.
24pub fn validate_edge_type(edge_type: &str) -> std::result::Result<(), EdgeTypeValidationError> {
25    if EDGE_TYPE_PATTERN.is_match(edge_type) {
26        Ok(())
27    } else {
28        Err(EdgeTypeValidationError(format!(
29            "Invalid edge_type '{edge_type}'. Must start with an uppercase letter, contain only \
30             uppercase letters, digits, underscores, and be 1-50 chars."
31        )))
32    }
33}
34
35/// Returns `true` if `edge_type` is valid.
36pub fn is_valid_edge_type(edge_type: &str) -> bool {
37    EDGE_TYPE_PATTERN.is_match(edge_type)
38}

40/// Standard, semantically-meaningful edge types.
41///
42/// All are UPPERCASE as required by the Neo4j-backed graph.
43#[non_exhaustive]
44pub struct EdgeType;
45
46impl EdgeType {
47    /// Ownership / grouping relationship.
48    pub const BELONGS_TO: &'static str = "BELONGS_TO";
49    /// Source was generated/created from target.
50    pub const CREATED_FROM: &'static str = "CREATED_FROM";
51    /// Soft reference relationship.
52    pub const REFERENCED: &'static str = "REFERENCED";
53    /// Source requires target to function.
54    pub const DEPENDS_ON: &'static str = "DEPENDS_ON";
55    /// Source was derived/modified from target.
56    pub const DERIVED_FROM: &'static str = "DERIVED_FROM";
57    /// Source contains/includes target.
58    pub const CONTAINS: &'static str = "CONTAINS";
59    /// Source replaces/supersedes target (belief revision).
60    pub const SUPERSEDES: &'static str = "SUPERSEDES";
61}

63/// Direction for edge queries and traversals.
64#[derive(Debug, Clone, Copy, PartialEq, Eq, Default)]
65pub enum EdgeDirection {
66    /// Edges where the queried revision is the source.
67    #[default]
68    Outgoing,
69    /// Edges where the queried revision is the target.
70    Incoming,
71    /// Edges in either direction.
72    Both,
73}
74
75impl EdgeDirection {
76    pub(crate) fn as_pb(self) -> i32 {
77        match self {
78            EdgeDirection::Outgoing => pb::EdgeDirection::Outgoing as i32,
79            EdgeDirection::Incoming => pb::EdgeDirection::Incoming as i32,
80            EdgeDirection::Both => pb::EdgeDirection::Both as i32,
81        }
82    }
83}

85/// A directed, typed relationship between two revisions.
86#[derive(Clone)]
87pub struct Edge {
88    /// Source revision reference.
89    pub source_kref: Kref,
90    /// Target revision reference.
91    pub target_kref: Kref,
92    /// Relationship type (see [`EdgeType`]).
93    pub edge_type: String,
94    /// Edge metadata.
95    pub metadata: std::collections::HashMap<String, String>,
96    /// Creation timestamp (ISO-8601), if set.
97    pub created_at: Option<String>,
98    /// Author user id.
99    pub author: String,
100    /// Author display name.
101    pub username: String,
102    client: Client,
103}
104
105impl Edge {
106    pub(crate) fn from_pb(pb: pb::Edge, client: Client) -> Self {
107        Edge {
108            source_kref: Kref::from_pb(&pb.source_kref.unwrap_or_default()),
109            target_kref: Kref::from_pb(&pb.target_kref.unwrap_or_default()),
110            edge_type: pb.edge_type,
111            metadata: pb.metadata,
112            created_at: (!pb.created_at.is_empty()).then_some(pb.created_at),
113            author: pb.author,
114            username: pb.username,
115            client,
116        }
117    }
118
119    /// Delete this edge.
120    pub async fn delete(&self) -> Result<()> {
121        self.client
122            .delete_edge(&self.source_kref, &self.target_kref, &self.edge_type)
123            .await
124    }
125}
126
127impl std::fmt::Debug for Edge {
128    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
129        write!(
130            f,
131            "Edge({} -> {} type={})",
132            self.source_kref, self.target_kref, self.edge_type
133        )
134    }
135}

170/// Result of a transitive edge traversal.
171pub struct TraversalResult {
172    /// Flat list of discovered revision references.
173    pub revision_krefs: Vec<Kref>,
174    /// Path details (populated when `include_path` was requested).
175    pub paths: Vec<RevisionPath>,
176    /// All traversed edges.
177    pub edges: Vec<Edge>,
178    /// Total number of discovered revisions.
179    pub total_count: i32,
180    /// True if results were limited by depth/limit.
181    pub truncated: bool,
182    client: Client,
183}
184
185impl TraversalResult {
186    pub(crate) fn new(
187        revision_krefs: Vec<Kref>,
188        paths: Vec<RevisionPath>,
189        edges: Vec<Edge>,
190        total_count: i32,
191        truncated: bool,
192        client: Client,
193    ) -> Self {
194        TraversalResult {
195            revision_krefs,
196            paths,
197            edges,
198            total_count,
199            truncated,
200            client,
201        }
202    }
203
204    /// Fetch full [`Revision`] objects for every discovered revision.
205    pub async fn get_revisions(&self) -> Result<Vec<Revision>> {
206        let mut out = Vec::with_capacity(self.revision_krefs.len());
207        for kref in &self.revision_krefs {
208            out.push(self.client.get_revision(kref.uri()).await?);
209        }
210        Ok(out)
211    }
212}
